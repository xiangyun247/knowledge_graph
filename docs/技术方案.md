## 一、设计原则

| 原则 | 说明 |
|------|------|
| **可演进** | 当前只用行为+问卷，EEG作为独立模块可随时插入 |
| **模块化** | 各模态独立采集、独立评估，最后统一融合 |
| **接口先行** | EEG的API和数据结构先定义好，硬件到了直接对接 |
| **向后兼容** | 不影响现有系统运行 |

### 4.1 统一评估请求/响应格式

```python
# backend/schemas/cognitive_load.py

from pydantic import BaseModel, Field
from typing import Optional, List, Literal
from datetime import datetime

class BehaviorFeatures(BaseModel):
    """行为特征 - 当前已实现"""
    task_id: str
    session_id: str
    source: Literal["patient_education", "chat", "medication_reminder"]
    duration_ms: int
    click_count: int
    back_count: int
    error_count: int
    step_count: int
    total_scroll_time_ms: int
    extra: dict = Field(default_factory=dict)


class QuestionnaireFeatures(BaseModel):
    """
    问卷特征 - 基于简化NASA-TLX

    简化NASA-TLX（6维度，无配对比较）：
    - 脑力需求 (mental_demand): 完成任务的脑力感知
    - 体力需求 (physical_demand): 完成任务的体力感知
    - 时间需求 (temporal_demand): 时间压力感知
    - 绩效 (performance): 对自己完成任务的满意度
    - 努力 (effort): 完成任务需要付出的努力
    - 挫败感 (frustration): 完成过程的烦躁程度

    每项 1-7 分，归一化到 0-1 后加权求均值
    """
    task_id: str
    source: str
    answers: List[dict]  # [{qid: str, value: int, dimension: str}]
    avg_score: float  # 1-7
    total_score: float  # 原始总分
    dimension_scores: dict  # {dimension: avg_score} 各维度平均分
    weighted_score: float = 0.0  # 加权评分（未来可配置权重）


class EEGFeatures(BaseModel):
    """EEG特征 - 预留接口"""
    session_id: str
    device_type: str  # "muse_s" / "openbci" / "emotiv"
    sampling_rate: int
    channels: int
    timestamp_start: int
    timestamp_end: int

    # 频段功率特征
    delta_power: float
    theta_power: float
    alpha_power: float
    beta_power: float
    gamma_power: float

    # 认知负荷相关比率
    theta_beta_ratio: float
    theta_alpha_ratio: float
    alpha_beta_ratio: float

    # 相对基线变化
    relative_delta: float = 0.0
    relative_theta: float = 0.0
    relative_alpha: float = 0.0
    relative_beta: float = 0.0
    relative_gamma: float = 0.0

    raw_score: float  # EEG单独评估分 0-1

    class Config:
        extra = "allow"  # 预留，允许后续扩展


class CognitiveLoadRequest(BaseModel):
    """多模态认知负荷评估请求"""
    user_id: str
    session_id: str
    task_id: str
    source: str

    # 必选：当前已有的模态
    behavior: Optional[BehaviorFeatures] = None
    questionnaire: Optional[QuestionnaireFeatures] = None

    # 预留：未来扩展
    eeg: Optional[EEGFeatures] = None
    eye_tracking: Optional[dict] = None  # 眼动预留
    gsr: Optional[dict] = None  # 皮肤电预留

    # 评估参数
    mode: Literal["realtime", "offline"] = "realtime"
    fusion_method: Literal["weighted", "model_fusion"] = "weighted"


class CognitiveLoadResponse(BaseModel):
    """评估响应 - 统一格式"""
    session_id: str
    task_id: str
    timestamp: datetime

    # 最终评分
    final_score: float = Field(..., ge=0, le=1)  # 0-1
    level: Literal["low", "medium", "high"]

    # 分模态贡献（透明可解释）
    breakdown: dict = Field(default_factory=dict)

    # 特征详情（用于调试和分析）
    features: dict = Field(default_factory=dict)

    # 建议
    recommendations: List[str] = []

    # 可用模态标志
    available_modalities: List[str] = []


class BaselineRequest(BaseModel):
    """个体基线采集请求 - 预留EEG校准接口"""
    user_id: str
    session_id: str
    modality: Literal["eeg", "behavior", "questionnaire"]
    baseline_data: dict
    calibration_type: str  # "eyes_closed", "eyes_open", "resting"


class BaselineResponse(BaseModel):
    baseline_id: str
    modality: str
    created_at: datetime
    status: str
```

## 五、融合评估引擎

### 5.1 融合调度器

```python
# backend/cognitive/fusion_orchestrator.py

from typing import Optional, Dict, List
from ..schemas.cognitive_load import (
    CognitiveLoadRequest, CognitiveLoadResponse,
    BehaviorFeatures, QuestionnaireFeatures, EEGFeatures
)
from .modalities.behavior_scorer import BehaviorScorer
from .modalities.questionnaire_scorer import QuestionnaireScorer
from .modalities.eeg_scorer import EEGScorer
from .fusion.weighted_fusion import WeightedFusion
from .fusion.model_fusion import ModelFusion

class FusionOrchestrator:
    """
    多模态融合调度器

    设计原则：
    - EEG scorer 存在但可为空（None），不影响融合
    - 根据available_modalities动态调整权重
    - 支持加权融合和模型融合两种模式
    """

    def __init__(self):
        self.behavior_scorer = BehaviorScorer()
        self.questionnaire_scorer = QuestionnaireScorer()
        self.eeg_scorer = EEGScorer()  # 预留，当前可能无设备

        # 默认权重（可配置）
        self.default_weights = {
            "behavior": 0.40,
            "questionnaire": 0.35,
            "eeg": 0.25
        }

        # 当前可用模态
        self.available_modalities = self._detect_available_modalities()

    def _detect_available_modalities(self) -> List[str]:
        """检测当前可用的模态"""
        available = []
        available.extend(["behavior", "questionnaire"])
        if self.eeg_scorer.is_available():
            available.append("eeg")
        return available

    def assess(self, request: CognitiveLoadRequest) -> CognitiveLoadResponse:
        """
        执行多模态认知负荷评估

        流程：
        1. 各模态独立评分
        2. 检测可用模态
        3. 计算动态权重
        4. 融合输出
        """
        breakdown = {}

        # 1. 行为评分
        behavior_score = None
        if request.behavior:
            behavior_score = self.behavior_scorer.score(request.behavior)
            breakdown["behavior"] = {
                "score": behavior_score,
                "weight": self.default_weights["behavior"],
                "available": True
            }
        else:
            breakdown["behavior"] = {"available": False}

        # 2. 问卷评分
        questionnaire_score = None
        if request.questionnaire:
            questionnaire_score = self.questionnaire_scorer.score(request.questionnaire)
            breakdown["questionnaire"] = {
                "score": questionnaire_score,
                "weight": self.default_weights["questionnaire"],
                "available": True
            }
        else:
            breakdown["questionnaire"] = {"available": False}

        # 3. EEG评分（预留）
        eeg_score = None
        eeg_available = self.eeg_scorer.is_available()
        if request.eeg and eeg_available:
            eeg_score = self.eeg_scorer.score(request.eeg)
            breakdown["eeg"] = {
                "score": eeg_score,
                "weight": self.default_weights["eeg"],
                "available": True
            }
        else:
            breakdown["eeg"] = {"available": False}

        # 4. 动态权重调整（根据可用模态）
        final_score, adjusted_breakdown = self._dynamic_fusion(
            breakdown, request.fusion_method
        )

        # 5. 生成建议
        recommendations = self._generate_recommendations(
            final_score, breakdown, request.source
        )

        return CognitiveLoadResponse(
            session_id=request.session_id,
            task_id=request.task_id,
            timestamp=datetime.now(),
            final_score=final_score,
            level=self._score_to_level(final_score),
            breakdown=adjusted_breakdown,
            features={"raw": breakdown},
            recommendations=recommendations,
            available_modalities=self._get_available_list(breakdown)
        )

    def _dynamic_fusion(self, breakdown: dict, method: str):
        """动态融合 - 根据可用模态调整权重和方法"""
        available = {k: v for k, v in breakdown.items() if v.get("available")}

        if len(available) == 0:
            return 0.5, breakdown

        if len(available) == 1:
            key = list(available.keys())[0]
            return available[key]["score"], breakdown

        if method == "weighted":
            fusion = WeightedFusion()
            return fusion.combine(available, self.default_weights)
        else:
            fusion = ModelFusion()
            return fusion.combine(available)

    def _score_to_level(self, score: float) -> str:
        if score < 0.4:
            return "low"
        elif score < 0.7:
            return "medium"
        else:
            return "high"

    def _get_available_list(self, breakdown: dict) -> List[str]:
        return [k for k, v in breakdown.items() if v.get("available")]
```

### 6.1 行为评分器

```python
# backend/cognitive/modalities/behavior_scorer.py

from typing import Dict
from ....schemas.cognitive_load import BehaviorFeatures

class BehaviorScorer:
    """
    行为特征评分器

    认知负荷行为指标：
    - 回退率(back_rate): 高 → 认知负荷高
    - 错误率(error_rate): 高 → 认知负荷高
    - 任务时长(duration): 过长/过短 → 可能负荷异常
    - 点击密度(click_density): 高 → 信息过载
    - 滚动次数(scroll_count): 反映内容复杂度感知
    """

    def score(self, behavior: BehaviorFeatures) -> float:
        """
        计算行为综合评分 (0-1)
        0 = 低认知负荷, 1 = 高认知负荷
        """
        metrics = {}

        # 1. 回退率
        if behavior.click_count > 0:
            back_rate = behavior.back_count / behavior.click_count
        else:
            back_rate = 0
        metrics["back_rate"] = back_rate

        # 2. 错误率
        if behavior.click_count > 0:
            error_rate = behavior.error_count / behavior.click_count
        else:
            error_rate = 0
        metrics["error_rate"] = error_rate

        # 3. 任务时长归一化（假设最佳区间10-60秒）
        duration_sec = behavior.duration_ms / 1000
        if duration_sec < 10:
            duration_norm = 0.3
        elif duration_sec <= 60:
            duration_norm = 0.2
        elif duration_sec <= 180:
            duration_norm = 0.5
        else:
            duration_norm = 0.8
        metrics["duration_norm"] = duration_norm

        # 4. 点击密度（每分钟点击）
        if behavior.duration_ms > 0:
            click_per_min = (behavior.click_count / behavior.duration_ms) * 60000
        else:
            click_per_min = 0
        metrics["click_density"] = min(click_per_min / 30, 1.0)

        # 5. 分步完成度
        if behavior.step_count > 0:
            completion_rate = min(behavior.click_count / behavior.step_count, 2) / 2
        else:
            completion_rate = 0.5
        metrics["completion_rate"] = completion_rate

        # 综合评分
        final = (
            metrics["back_rate"] * 0.30 +
            metrics["error_rate"] * 0.25 +
            metrics["duration_norm"] * 0.20 +
            metrics["click_density"] * 0.15 +
            (1 - metrics["completion_rate"]) * 0.10
        )

        return min(max(final, 0), 1)
```

### 6.2 问卷评分器

```python
# backend/cognitive/modalities/questionnaire_scorer.py

from typing import List, Dict
from ....schemas.cognitive_load import QuestionnaireFeatures

class NASA_TLX_Scorer:
    """
    简化NASA-TLX评分器

    理论基础：
    - NASA-TLX是主观工作负荷评估的金标准
    - 原始版本包含20步配对比较（确定各维度权重），过于复杂
    - 简化版直接使用6个维度等权重评分，适合老年人/快速评估

    评分维度（每项1-7分）：
    - mental_demand: 脑力需求（需要多少脑力思考？）
    - physical_demand: 体力需求（需要多少体力？）
    - temporal_demand: 时间需求（时间够用吗？）
    - performance: 绩效感知（对自己的表现满意吗？）
    - effort: 努力程度（需要多少努力才能完成？）
    - frustration: 挫败感（完成过程让你烦躁吗？）

    认知负荷解释：
    - 分数越高 = 主观认知负荷越高
    - 各维度独立计算，可做雷达图分析
    """

    # 各维度默认权重（等权重）
    DEFAULT_WEIGHTS = {
        "mental_demand": 1.0,
        "physical_demand": 1.0,
        "temporal_demand": 1.0,
        "performance": 1.0,
        "effort": 1.0,
        "frustration": 1.0
    }

    def __init__(self, weights: Dict[str, float] = None):
        """
        初始化评分器

        Args:
            weights: 各维度权重，若为None则使用等权重
        """
        self.weights = weights or self.DEFAULT_WEIGHTS

    def score(self, questionnaire: QuestionnaireFeatures) -> float:
        """
        计算问卷评分 (0-1)

        公式：
        1. 将1-7分归一化到0-1: (raw - 1) / 6
        2. 按维度分组计算各维度平均分
        3. 各维度加权平均

        Returns:
            0-1之间的认知负荷分数（越高负荷越高）
        """
        if not questionnaire.answers:
            return 0.5

        # 按维度分组计算平均分
        dimension_sums = {}
        dimension_counts = {}

        for answer in questionnaire.answers:
            dim = answer.get("dimension", "mental_demand")
            value = answer.get("value", 1)

            if dim not in dimension_sums:
                dimension_sums[dim] = 0.0
                dimension_counts[dim] = 0

            dimension_sums[dim] += value
            dimension_counts[dim] += 1

        # 计算各维度归一化分数
        normalized_scores = []
        total_weight = 0.0

        for dim, weight in self.weights.items():
            if dim in dimension_sums:
                avg = dimension_sums[dim] / dimension_counts[dim]
                # 归一化到0-1: (value - 1) / 6
                normalized = (avg - 1) / 6
                normalized_scores.append(normalized * weight)
                total_weight += weight

        if total_weight == 0:
            return 0.5

        # 加权平均
        final_score = sum(normalized_scores) / total_weight

        return min(max(final_score, 0), 1)

    def get_dimension_scores(self, questionnaire: QuestionnaireFeatures) -> Dict[str, float]:
        """
        获取各维度详细评分（用于雷达图分析）

        Returns:
            {dimension: normalized_score} 各维度归一化分数
        """
        dimension_scores = {}

        for answer in questionnaire.answers:
            dim = answer.get("dimension", "mental_demand")
            value = answer.get("value", 1)

            if dim not in dimension_scores:
                dimension_scores[dim] = []
            dimension_scores[dim].append(value)

        # 计算每个维度的平均分
        result = {}
        for dim, values in dimension_scores.items():
            avg = sum(values) / len(values)
            result[dim] = (avg - 1) / 6  # 归一化到0-1

        return result

    def interpret_score(self, score: float) -> Dict[str, any]:
        """
        解释评分结果

        Args:
            score: 0-1的认知负荷分数

        Returns:
            {level, description, suggestions} 解释和建议
        """
        if score < 0.33:
            level = "low"
            description = "低认知负荷"
            suggestions = [
                "任务对用户来说较为轻松",
                "可以考虑增加内容复杂度或减少辅助",
                "适合新手引导或轻度认知任务"
            ]
        elif score < 0.66:
            level = "medium"
            description = "中等认知负荷"
            suggestions = [
                "任务难度适中，符合适度的挑战",
                "继续保持当前的信息呈现方式",
                "关注用户的疲劳积累趋势"
            ]
        else:
            level = "high"
            description = "高认知负荷"
            suggestions = [
                "任务对用户来说较为吃力",
                "建议简化信息呈现（如分步、减少文字）",
                "考虑增加辅助提示或降低内容难度"
            ]

        return {
            "level": level,
            "description": description,
            "suggestions": suggestions,
            "raw_score": score
        }
```

```python
# backend/cognitive/report_generator.py

from typing import List, Optional, Dict
from dataclasses import dataclass
from datetime import datetime, timedelta
from enum import Enum

class ReportType(Enum):
    SINGLE = "single"      # 单次评估报告
    TASK = "task"          # 任务分析报告
    TREND = "trend"        # 趋势分析报告
    PERIOD = "period"      # 周期汇总报告

@dataclass
class ReportConfig:
    """报告生成配置"""
    report_type: ReportType
    user_id: str
    assessment_id: Optional[int] = None
    task_id: Optional[str] = None
    session_id: Optional[str] = None
    period_days: int = 7  # 趋势/周期报告的天数
    include_ai_suggestions: bool = False  # 是否启用AI建议


class CognitiveReportGenerator:
    """
    认知负荷评估报告生成器

    职责：
    1. 根据报告类型聚合数据
    2. 计算评分、趋势、对比基准
    3. 生成结构化报告
    4. 可选：调用LLM生成个性化建议
    """

    def __init__(self, repository, nasa_scorer, behavior_scorer):
        self.repository = repository
        self.nasa_scorer = nasa_scorer
        self.behavior_scorer = behavior_scorer

        # NASA-TLX维度中文映射
        self.dimension_names = {
            "mental_demand": "脑力需求",
            "physical_demand": "体力需求",
            "temporal_demand": "时间需求",
            "performance": "绩效感知",
            "effort": "努力程度",
            "frustration": "挫败感"
        }

    async def generate(self, config: ReportConfig) -> Dict:
        """
        生成评估报告

        Args:
            config: 报告配置

        Returns:
            结构化报告字典
        """
        if config.report_type == ReportType.SINGLE:
            return await self._generate_single_report(config)
        elif config.report_type == ReportType.TASK:
            return await self._generate_task_report(config)
        elif config.report_type == ReportType.TREND:
            return await self._generate_trend_report(config)
        elif config.report_type == ReportType.PERIOD:
            return await self._generate_period_report(config)

    async def _generate_single_report(self, config: ReportConfig) -> Dict:
        """
        生成单次评估报告

        内容：
        - 综合评分 + 等级
        - NASA-TLX雷达图数据
        - 关键发现
        - 即时建议
        """
        # 1. 获取评估数据
        assessment = await self.repository.get_assessment(config.assessment_id)
        nasa_answers = await self.repository.get_nasa_tlx_answers(config.assessment_id)

        # 2. 计算雷达图数据
        radar_data = self._compute_radar_chart(nasa_answers)

        # 3. 生成关键发现
        key_findings = self._extract_key_findings(assessment, nasa_answers)

        # 4. 生成建议
        suggestions = self._generate_suggestions(assessment, nasa_answers)

        # 5. 构建报告
        report = {
            "report_type": "single",
            "user_id": config.user_id,
            "assessment_id": config.assessment_id,
            "task_id": assessment["task_id"],
            "source": assessment["source"],

            # 评分汇总
            "summary": {
                "overall_score": assessment["final_score"],
                "overall_level": assessment["level"],
                "behavior_score": assessment.get("behavior_score"),
                "questionnaire_score": assessment.get("questionnaire_score"),
                "duration_ms": assessment.get("duration_ms"),
                "task_time": self._format_duration(assessment.get("duration_ms", 0))
            },

            # NASA-TLX雷达图
            "radar_chart": radar_data,

            # 关键发现
            "key_findings": key_findings,

            # 风险预警
            "risk_alerts": self._generate_risk_alerts(assessment, nasa_answers),

            # 积极方面
            "positive_aspects": self._extract_positive_aspects(assessment, nasa_answers),

            # 个性化建议
            "suggestions": suggestions,

            # 参考基准
            "benchmarks": await self._compute_benchmarks(config.user_id, assessment),

            # 生成时间
            "generated_at": datetime.now().isoformat()
        }

        # 6. AI增强建议（可选）
        if config.include_ai_suggestions:
            report["ai_suggestions"] = await self._generate_ai_suggestions(report)

        return report

    async def _generate_trend_report(self, config: ReportConfig) -> Dict:
        """
        生成趋势分析报告

        内容：
        - 认知负荷时序变化
        - 各维度趋势对比
        - 高峰时段分析
        - 长期建议
        """
        # 1. 获取历史评估数据
        assessments = await self.repository.get_user_assessments(
            user_id=config.user_id,
            days=config.period_days
        )

        if not assessments:
            return {"error": "数据不足，无法生成趋势报告", "required": "至少需要3次评估"}

        # 2. 计算时序数据
        timeline = self._compute_timeline(assessments)

        # 3. 计算NASA-TLX趋势
        nasa_trends = await self._compute_nasa_tlx_trends(
            config.user_id, config.period_days
        )

        # 4. 趋势判断
        trend_direction, change_rate = self._analyze_trend(timeline["scores"])

        # 5. 高峰时段分析
        peak_times = self._analyze_peak_times(assessments)

        # 6. 趋势判断与建议
        trend_summary = {
            "trend_direction": trend_direction,
            "change_rate": change_rate,
            "avg_score": sum(timeline["scores"]) / len(timeline["scores"]),
            "score_variance": self._compute_variance(timeline["scores"]),
            "peak_times": peak_times,
            "lowest_load_periods": self._find_lowest_periods(assessments)
        }

        report = {
            "report_type": "trend",
            "user_id": config.user_id,
            "period_days": config.period_days,
            "assessment_count": len(assessments),

            # 时序趋势
            "timeline": timeline,

            # NASA-TLX各维度趋势
            "nasa_tlx_trends": nasa_trends,

            # 趋势汇总
            "trend_summary": trend_summary,

            # 维度雷达图对比（最新 vs 平均）
            "radar_comparison": {
                "latest": self._compute_latest_radar(assessments),
                "average": self._compute_average_radar(assessments),
                "baseline": await self._get_user_baseline(config.user_id)
            },

            # 风险预警
            "risk_alerts": self._generate_trend_risk_alerts(trend_summary),

            # 趋势建议
            "suggestions": self._generate_trend_suggestions(trend_summary),

            # 生成时间
            "generated_at": datetime.now().isoformat()
        }

        return report

    def _compute_radar_chart(self, nasa_answers: Dict) -> Dict:
        """
        计算NASA-TLX雷达图数据

        Returns:
            {
                "dimensions": ["脑力需求", "体力需求", ...],
                "scores": [0.67, 0.17, 0.33, 0.50, 0.67, 0.33],  # 归一化到0-1
                "raw_scores": [4.2, 2.1, 2.9, 3.5, 4.5, 2.8],  # 原始1-7分
                "norm_values": [0.60, 0.18, 0.32, 0.50, 0.70, 0.30],  # 老年人常模
            }
        """
        dimensions = list(self.dimension_names.keys())
        dimension_labels = list(self.dimension_names.values())

        scores = []
        raw_scores = []

        for dim in dimensions:
            raw_val = nasa_answers.get(dim, 4)  # 默认中等
            raw_scores.append(raw_val)
            normalized = (raw_val - 1) / 6  # 归一化到0-1
            scores.append(round(normalized, 3))

        return {
            "dimensions": dimension_labels,
            "scores": scores,
            "raw_scores": raw_scores,
            "norm_values": self._get_elderly_norm_values()
        }

    def _extract_key_findings(self, assessment: Dict, nasa_answers: Dict) -> List[str]:
        """
        从评估数据中提取关键发现

        Returns:
            ["回退率为15%，略高于正常水平", "NASA-TLX中脑力需求维度得分最高", ...]
        """
        findings = []

        # 行为发现
        behavior_features = assessment.get("behavior_features", {})
        if behavior_features:
            back_rate = behavior_features.get("back_rate", 0)
            if back_rate > 0.2:
                findings.append(f"回退率为{back_rate*100:.0f}%，高于正常水平，可能遇到理解困难")
            elif back_rate > 0.1:
                findings.append(f"回退率为{back_rate*100:.0f}%，处于正常范围")

            error_rate = behavior_features.get("error_rate", 0)
            if error_rate > 0.1:
                findings.append(f"错误操作率为{error_rate*100:.0f}%，建议简化操作流程")

        # NASA-TLX发现
        max_dim, max_val = self._find_max_dimension(nasa_answers)
        if max_dim and max_val > 4:
            dim_name = self.dimension_names.get(max_dim, max_dim)
            findings.append(f"NASA-TLX中{dim_name}维度得分最高({max_val}/7)，是主要负荷来源")

        # 任务时长发现
        duration_ms = assessment.get("duration_ms", 0)
        duration_sec = duration_ms / 1000
        if duration_sec > 180:
            findings.append(f"任务完成耗时{duration_sec/60:.1f}分钟较长，可能内容偏多或难度偏高")
        elif duration_sec < 15:
            findings.append(f"任务完成较快({duration_sec:.0f}秒)，内容相对简单")

        return findings

    def _generate_risk_alerts(self, assessment: Dict, nasa_answers: Dict) -> List[str]:
        """
        生成风险预警

        预警规则：
        - 综合评分 > 0.7 → 高认知负荷预警
        - 挫败感 > 5 → 高挫败感预警
        - 回退率 > 0.3 → 操作困难预警
        - 任务时长 > 5分钟 → 内容过载预警
        """
        alerts = []

        # 综合评分预警
        if assessment.get("final_score", 0) > 0.7:
            alerts.append("⚠️ 高认知负荷预警：当前任务对用户造成较高认知负荷")

        # 挫败感预警
        frustration = nasa_answers.get("frustration", 0)
        if frustration >= 5:
            alerts.append("⚠️ 高挫败感预警：用户在任务中感到明显烦躁")

        # 行为预警
        behavior = assessment.get("behavior_features", {})
        if behavior.get("back_rate", 0) > 0.3:
            alerts.append("⚠️ 操作困难预警：频繁回退表明内容可能较难理解")

        if behavior.get("error_rate", 0) > 0.15:
            alerts.append("⚠️ 错误率高预警：操作错误频繁，需检查界面设计")

        # 任务时长预警
        if assessment.get("duration_ms", 0) > 300000:  # 5分钟
            alerts.append("⚠️ 内容过载预警：任务时间过长，建议拆分内容")

        return alerts

    def _generate_suggestions(self, assessment: Dict, nasa_answers: Dict) -> List[Dict]:
        """
        生成个性化建议

        建议结构：
        {
            "category": "界面优化",
            "priority": "high/medium/low",
            "suggestion": "建议内容...",
            "expected_impact": "可降低XX%认知负荷",
            "source": "基于NASA-TLX脑力需求分析"
        }
        """
        suggestions = []

        # 1. 基于NASA-TLX的建议
        mental = nasa_answers.get("mental_demand", 0)
        if mental >= 5:
            suggestions.append({
                "category": "内容优化",
                "priority": "high",
                "suggestion": "建议将长段落内容拆分为更小的步骤，每步聚焦一个要点",
                "expected_impact": "可降低20-30%脑力负荷",
                "source": "NASA-TLX脑力需求维度分析"
            })
            suggestions.append({
                "category": "呈现方式",
                "priority": "high",
                "suggestion": "使用通俗语言，避免专业术语，必要时提供通俗解释",
                "expected_impact": "可降低15-25%理解难度",
                "source": "NASA-TLX脑力需求维度分析"
            })

        frustration = nasa_answers.get("frustration", 0)
        if frustration >= 4:
            suggestions.append({
                "category": "用户体验",
                "priority": "high",
                "suggestion": "增加进度指示器，让用户知道当前进度和剩余时间",
                "expected_impact": "可降低焦虑感和挫败感",
                "source": "NASA-TLX挫败感维度分析"
            })

        temporal = nasa_answers.get("temporal_demand", 0)
        if temporal >= 5:
            suggestions.append({
                "category": "时间安排",
                "priority": "medium",
                "suggestion": "避免在用户疲劳时段（如下午3点、晚饭后）安排复杂任务",
                "expected_impact": "可减少15-20%时间压力感",
                "source": "NASA-TLX时间需求维度分析"
            })

        # 2. 基于行为的建议
        behavior = assessment.get("behavior_features", {})
        if behavior.get("back_rate", 0) > 0.2:
            suggestions.append({
                "category": "导航优化",
                "priority": "medium",
                "suggestion": "增加'上一步'确认提示，减少用户回退需求",
                "expected_impact": "可减少30-40%回退操作",
                "source": "回退率行为分析"
            })

        if behavior.get("click_density", 0) > 0.8:
            suggestions.append({
                "category": "界面简化",
                "priority": "medium",
                "suggestion": "减少界面元素数量，聚焦核心操作",
                "expected_impact": "可降低视觉搜索负担",
                "source": "点击密度行为分析"
            })

        # 3. 按优先级排序
        priority_order = {"high": 0, "medium": 1, "low": 2}
        suggestions.sort(key=lambda x: priority_order.get(x["priority"], 2))

        return suggestions[:5]  # 最多返回5条建议

    def _compute_benchmarks(self, user_id: str, assessment: Dict) -> Dict:
        """
        计算对比基准

        对比维度：
        - 用户历史平均
        - 同龄组平均（老年人常模）
        - 用户历史最佳/最差
        """
        historical = self.repository.get_user_historical_stats(user_id)

        return {
            "user_historical_avg": historical.get("avg_score", 0.5),
            "user_historical_best": historical.get("min_score", 0),
            "user_historical_worst": historical.get("max_score", 1),
            "elderly_norm": 0.55,  # 老年人群体常模
            "percentile": self._compute_percentile(
                assessment["final_score"],
                historical.get("avg_score", 0.5)
            )
        }

    def _analyze_trend(self, scores: List[float]) -> tuple:
        """
        分析趋势方向和变化率

        Returns:
            (trend_direction: str, change_rate: float)
            trend_direction: "increasing" / "decreasing" / "stable"
        """
        if len(scores) < 2:
            return "stable", 0.0

        # 简单线性回归斜率
        n = len(scores)
        x_mean = (n - 1) / 2
        y_mean = sum(scores) / n

        numerator = sum((i - x_mean) * (scores[i] - y_mean) for i in range(n))
        denominator = sum((i - x_mean) ** 2 for i in range(n))

        if denominator == 0:
            return "stable", 0.0

        slope = numerator / denominator

        # 标准化变化率（相对于均值）
        if y_mean > 0:
            change_rate = slope / y_mean
        else:
            change_rate = 0.0

        # 判断趋势
        if change_rate > 0.1:
            trend = "increasing"
        elif change_rate < -0.1:
            trend = "decreasing"
        else:
            trend = "stable"

        return trend, round(change_rate, 3)
```

## 九、数据库与持久化设计

### 9.1 设计原则

| 原则 | 说明 |
|------|------|
| **事务性** | 评估结果一次性写入，保证行为+问卷+评分的一致性 |
| **可追溯** | 原始事件完整存储，支持回溯和重评分 |
| **扩展性** | EEG/眼动等新模态可直接新增字段，不破坏结构 |
| **查询友好** | 常用查询字段建立索引，支持趋势分析 |
| **分层存储** | 热数据（近期评估） vs 冷数据（历史归档） |

### 9.4 数据操作API

```python
# backend/cognitive/repository.py
# 数据持久化仓储层

from typing import List, Optional
from datetime import datetime
from sqlalchemy import select, func, and_
from sqlalchemy.ext.asyncio import AsyncSession

class CognitiveLoadRepository:
    """认知负荷数据仓储层"""

    def __init__(self, session: AsyncSession):
        self.session = session

    async def save_assessment(self, assessment_data: dict) -> int:
        """
        保存评估记录

        Args:
            assessment_data: 评估数据字典

        Returns:
            assessment_id: 新增记录的ID
        """
        # 1. 保存行为事件
        if assessment_data.get("behavior_events"):
            await self._save_behavior_events(
                assessment_data["user_id"],
                assessment_data["session_id"],
                assessment_data["task_id"],
                assessment_data["source"],
                assessment_data["behavior_events"]
            )

        # 2. 保存NASA-TLX答案
        nasa_tlx_id = None
        if assessment_data.get("nasa_tlx_answers"):
            nasa_tlx_id = await self._save_nasa_tlx_answers(
                assessment_data["nasa_tlx_answers"]
            )

        # 3. 保存评估主记录
        assessment_id = await self._save_assessment_record(
            assessment_data, nasa_tlx_id
        )

        # 4. 生成评估报告
        await self._generate_report(assessment_id, assessment_data)

        # 5. 更新用户画像
        await self._update_user_profile(assessment_data["user_id"])

        await self.session.commit()
        return assessment_id

    async def get_user_assessments(
        self,
        user_id: str,
        source: Optional[str] = None,
        start_date: Optional[datetime] = None,
        end_date: Optional[datetime] = None,
        limit: int = 100,
        offset: int = 0
    ) -> List[dict]:
        """
        获取用户的评估历史

        支持分页、来源筛选、时间范围筛选
        """
        query = select(cognitive_load_assessment).where(
            cognitive_load_assessment.user_id == user_id
        )

        if source:
            query = query.where(cognitive_load_assessment.source == source)

        if start_date:
            query = query.where(
                cognitive_load_assessment.created_at >= start_date
            )

        if end_date:
            query = query.where(
                cognitive_load_assessment.created_at <= end_date
            )

        query = query.order_by(
            cognitive_load_assessment.created_at.desc()
        ).limit(limit).offset(offset)

        result = await self.session.execute(query)
        return [row.to_dict() for row in result.scalars()]

    async def get_task_behavior_events(
        self,
        task_id: str,
        session_id: str
    ) -> List[dict]:
        """获取某任务的所有行为事件(用于重评分)"""
        query = select(cognitive_behavior_events).where(
            and_(
                cognitive_behavior_events.task_id == task_id,
                cognitive_behavior_events.session_id == session_id
            )
        ).order_by(cognitive_behavior_events.ts)

        result = await self.session.execute(query)
        return [row.to_dict() for row in result.scalars()]

    async def get_trend_analysis(
        self,
        user_id: str,
        days: int = 7,
        source: Optional[str] = None
    ) -> dict:
        """
        获取认知负荷趋势分析

        Returns:
            {
                "dates": ["2024-01-01", "2024-01-02", ...],
                "scores": [0.45, 0.52, ...],
                "levels": ["low", "medium", ...],
                "nasa_tlx_trends": {
                    "mental_demand": [0.5, 0.6, ...],
                    ...
                },
                "summary": {
                    "avg_score": 0.48,
                    "trend": "stable",
                    "peak_time": "下午3点"
                }
            }
        """
        # 查询最近N天的评估数据
        query = select(cognitive_load_assessment).where(
            and_(
                cognitive_load_assessment.user_id == user_id,
                cognitive_load_assessment.created_at >= func.date_sub(
                    func.now(), days
                )
            )
        )

        if source:
            query = query.where(cognitive_load_assessment.source == source)

        result = await self.session.execute(query)
        assessments = [row.to_dict() for row in result.scalars()]

        return self._compute_trend_analysis(assessments, days)

    async def recalculate_assessment(self, assessment_id: int) -> dict:
        """
        重新计算评估评分

        当行为评分规则更新时，可重新计算历史评估
        """
        # 1. 获取原始行为事件
        assessment = await self.get_assessment_by_id(assessment_id)
        behavior_events = await self.get_task_behavior_events(
            assessment["task_id"],
            assessment["session_id"]
        )

        # 2. 重新计算行为评分
        behavior_features = self.behavior_scorer.compute_features(behavior_events)
        new_behavior_score = self.behavior_scorer.score(behavior_features)

        # 3. 更新评估记录
        await self.session.execute(
            update(cognitive_load_assessment).where(
                cognitive_load_assessment.id == assessment_id
            ).values(
                behavior_score=new_behavior_score,
                behavior_features=behavior_features,
                final_score=self._compute_final_score(...)
            )
        )

        return {"status": "updated", "new_behavior_score": new_behavior_score}
```

```javascript
// src/utils/cognitiveLoadEnhanced.js
// 增强版认知负荷评估（支持多模态）

import { recordEvent, recordQuestionnaire, getCognitiveEvents } from './cognitiveLoad'
import { request } from './request'

class CognitiveLoadAssessor {
  constructor() {
    this.orchestrator = null
  }

  async init() {
    const resp = await request.get('/api/cognitive-load/modalities')
    this.availableModalities = resp.data.available
  }

  async assessTask(taskId, source, questionnaireAnswers) {
    // 1. 收集行为事件
    const behaviorEvents = getCognitiveEvents().filter(
      e => e.task_id === taskId
    )

    // 2. 构建行为特征
    const behaviorFeatures = this._computeBehaviorFeatures(behaviorEvents, taskId)

    // 3. 构建问卷特征
    const questionnaireFeatures = {
      task_id: taskId,
      source,
      answers: questionnaireAnswers,
      avg_score: this._computeAvgScore(questionnaireAnswers)
    }

    // 4. 调用评估API
    const response = await request.post('/api/cognitive-load/assess', {
      user_id: this.getCurrentUserId(),
      session_id: this.getCurrentSessionId(),
      task_id: taskId,
      source,
      behavior: behaviorFeatures,
      questionnaire: questionnaireFeatures,
    })

    return response.data
  }

  _computeBehaviorFeatures(events, taskId) {
    const taskEvents = events.filter(e => e.task_id === taskId)
    return {
      task_id: taskId,
      session_id: this.getCurrentSessionId(),
      source: taskEvents[0]?.source,
      duration_ms: this._computeDuration(taskEvents),
      click_count: taskEvents.filter(e => e.event_type === 'click').length,
      back_count: taskEvents.filter(e => e.event_type === 'back').length,
      error_count: taskEvents.filter(e => e.event_type === 'error_or_repeat').length,
      step_count: taskEvents.filter(e => e.event_type === 'step_view').length,
      total_scroll_time_ms: 0
    }
  }
}
```
